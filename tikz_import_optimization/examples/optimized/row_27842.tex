\documentclass[border=2mm]{standalone}
\usepackage{tikz}
\usepackage{tikz-3dplot}

% Define a 3D bar pic with customizable dimensions and face styles
\tikzset{
  pics/3d bar/.style={
    code={
      \tikzset{3d bar/.cd, #1}
      % Draw x-facing side (right/left)
      \path[3d bar/x face] 
        (\mydx/2,\mydy/2,0) -- (\mydx/2,\mydy/2,\myh)
        -- (-\mydx/2,\mydy/2,\myh) -- (-\mydx/2,\mydy/2,0) -- cycle;
      % Draw y-facing side (front/back)
      \path[3d bar/y face] 
        (\mydx/2,\mydy/2,0) -- (\mydx/2,\mydy/2,\myh)
        -- (\mydx/2,-\mydy/2,\myh) -- (\mydx/2,-\mydy/2,0) -- cycle;
      % Draw top (z) face
      \path[3d bar/z face] 
        (\mydx/2,\mydy/2,\myh) -- (-\mydx/2,\mydy/2,\myh)
        -- (-\mydx/2,-\mydy/2,\myh) -- (\mydx/2,-\mydy/2,\myh) -- cycle;
    }
  },
  3d bar/.cd,
  dx/.store in=\mydx, dx=1,
  dy/.store in=\mydy, dy=1,
  h/.store in=\myh, h=1,
  x face/.style={draw=blue!50, fill=cyan!20},
  y face/.style={draw=blue!50, fill=cyan!50},
  z face/.style={draw=blue!50, fill=cyan!30}
}

\begin{document}
\tdplotsetmaincoords{70}{110}
\begin{tikzpicture}[tdplot_main_coords]

% First plot: coarse grid (n=5)
\begin{scope}[
    declare function={
        f(\x,\y) = 1 + 3*exp(-\x/5 - \y/4); % surface function
        n = 5;                               % grid resolution
        xmin = 0; xmax = 5;
        ymin = 0; ymax = 5;
    }
  ]
  \pgfmathtruncatemacro{\myn}{n}
  % Set bar base dimensions based on grid cell size
  \tikzset{
    3d bar/dx/.evaluated = {(xmax - xmin)/n},
    3d bar/dy/.evaluated = {(ymax - ymin)/n}
  }
  % Iterate over grid cells (i: x-direction, j: y-direction)
  \foreach \i in {1,...,\myn} {
    \foreach \j in {1,...,\myn} {
      % Compute center of current grid cell
      \pgfmathsetmacro{\myx}{xmin + (\i - 0.5)*(xmax - xmin)/\myn}
      \pgfmathsetmacro{\myy}{ymin + (\j - 0.5)*(ymax - ymin)/\myn}
      \pgfmathsetmacro{\myf}{f(\myx,\myy)}
      % Place 3D bar at computed position with height = f(x,y)
      \path (\myx, \myy, 0) pic {3d bar={h=\myf}};
    }
  }
\end{scope}

% Second plot: fine grid (n=20), shifted right
\begin{scope}[
    xshift=7cm,
    declare function={
        f(\x,\y) = 0.5 + 3*exp(-\x/5 - \y/4); % slightly lower base
        n = 20;                                % higher resolution
        xmin = 0; xmax = 5;
        ymin = 0; ymax = 5;
    }
  ]
  \pgfmathtruncatemacro{\myn}{n}
  \tikzset{
    3d bar/dx/.evaluated = {(xmax - xmin)/n},
    3d bar/dy/.evaluated = {(ymax - ymin)/n}
  }
  \foreach \i in {1,...,\myn} {
    \foreach \j in {1,...,\myn} {
      \pgfmathsetmacro{\myx}{xmin + (\i - 0.5)*(xmax - xmin)/\myn}
      \pgfmathsetmacro{\myy}{ymin + (\j - 0.5)*(ymax - ymin)/\myn}
      \pgfmathsetmacro{\myf}{f(\myx,\myy)}
      \path (\myx, \myy, 0) pic {3d bar={h=\myf}};
    }
  }
\end{scope}

\end{tikzpicture}
\end{document}
